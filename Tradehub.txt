create database tradehub;
use tradehub;
drop table if exists picture;
drop table if exists accountdata;

create table accountdata(
	account_id int unsigned not null auto_increment,
	account_username varchar(50) not null,
	account_password varchar(50) not null, 
	account_email varchar(50) not null,
	
	constraint account_id_PK primary key (account_id)
);

create table picture(
	picture_id int unsigned not null auto_increment,
	picture_name varchar(100) not null,	
	picture_ending varchar(30) not null, 
	picture_description varchar(100) null,

	constraint picture_id_PK primary key (picture_id),
	constraint picture_id_FK foreign key (picture_id) references accountdata(account_id)
);
